This example demonstrates how to use the ChatGPT Automation API in your application.

Code Example

// Initialize the API
const api = await createChatGPTAPI();

// Send a prompt
try {
  const response = await api.sendPrompt("Your prompt here", {
    onProgress: (status) => {
      console.log("Status:", status);
    }
  });
  console.log("Response:", response);
} catch (error) {
  console.error("Error:", error.message);
}
